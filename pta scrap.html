import requests as requ
from bs4 import BeautifulSoup as bso
from datetime import datetime
import csv
import pandas as pd

hades = {'user-agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/106.0.0.0 Safari/537.36'}

url = 'https://pta.trunojoyo.ac.id/welcome/index/'

headers = {
    'user-agent': 'Mozilla/5.0 (Linux; Android 6.0; Nexus 5 Build/MRA58N) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/106.0.0.0 Mobile Safari/537.36'
}

temp = []
for page in range(1, 50):
  req = requ.get(url+str(page), headers=headers)
  bs = bso(req.text, 'html.parser')
  items = bs.findAll('li', {'data-id':'id-1'})
  for item in items:
    # temp = []
    title = item.find('a', 'title').text
    writer = item.find('span', '').text.replace('Penulis :', '')
    abstrac = item.find('p', {'align': 'justify'}).text
    temp.append([title, writer, abstrac])
print(temp)
