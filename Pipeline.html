data = orchest.get_inputs()  # data = [(df_data, df_target)]

X_train, X_test, y_train, y_test = train_test_split(X.values, y.values, test_size=0.20)

Eksploratory Data#

import pandas as pd
from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
import orchest

# get the input data
data = orchest.get_inputs()

# split the input data into X and y
data, target = data["data"]
X = data.values
y = target.values
# define the preprocessing steps
preprocessing_steps = [
    StandardScaler(),
]

# fit and transform the preprocessing steps to X
for step in preprocessing_steps:
    X = step.fit_transform(X)

# define the PCA model with n_components = 2
pca = PCA(n_components=2)

# fit and transform PCA to X
X_pca = pca.fit_transform(X)

# output the transformed data
orchest.output((X_pca, y), name="pca_data")

import numpy as np
import orchest
from sklearn.metrics import accuracy_score
from sklearn.neighbors import KNeighborsClassifier

data = orchest.get_inputs()
X_train, y_train, X_test, y_test = data["training_data"]

print("Fitting the model...")
# Fit model.
model = KNeighborsClassifier()
model.fit(X_train, y_train)

y_test_prediction = model.predict(X_test)
test_accuracy = accuracy_score(y_test, y_test_prediction)

# Output the accuracy for later use in the pipeline.
orchest.output(test_accuracy, name="knn-classifier-accuracy")

import numpy as np
import orchest
from sklearn.metrics import accuracy_score
from sklearn.naive_bayes import GaussianNB

data = orchest.get_inputs()
X_train, y_train, X_test, y_test = data["training_data"]

model = GaussianNB()
model.fit(X_train, y_train.ravel())

# Make predictions on the test set
y_test_prediction = model.predict(X_test)

# Calculate the accuracy score
test_accuracy = accuracy_score(y_test, y_test_prediction)

# Save results
orchest.output(test_accuracy, name="naive-bayes-accuracy")
