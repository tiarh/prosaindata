import numpy as np

def split_sequence(sequence, n_steps, n_features):
    X, y = [], []
    for i in range(len(sequence)):
        # Mencari index akhir dari sequence yang ingin dipisahkan
        end_ix = i + n_steps
        # Memastikan index akhir tidak melebihi panjang sequence
        if end_ix > len(sequence)-1:
            break
        # Mendapatkan input dan output untuk sequence saat ini
        seq_x, seq_y = sequence[i:end_ix, :-1], sequence[end_ix, -1]
        X.append(seq_x)
        y.append(seq_y)
    # Mengubah list menjadi array numpy
    X = np.array(X)
    y = np.array(y)
    return X, y
